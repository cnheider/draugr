def python_seed(s: int = 2318) -> None:
    """ """
    random.seed(s)
    os.environ["PYTHONHASHSEED"] = str(s)


def numpy_seed(s: int = 78213) -> None:
    """ """
    import numpy

    numpy.random.seed(s)
