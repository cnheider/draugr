[docs]def minmax_to_xywh_torch(boxes: torch.Tensor) -> torch.tensor:
    """

    :param boxes:
    :type boxes:
    :return:
    :rtype:"""
    xmin, ymin, xmax, ymax = boxes.unbind(1)
    return torch.stack((xmin, ymin, xmax - xmin, ymax - ymin), dim=1)
