[docs]def float_chw_to_hwc_uint(
    inp: numpy.ndarray, *, unnormalise: bool = True
) -> numpy.ndarray:
    """

    :param inp:
    :type inp:
    :param unnormalise:
    :type unnormalise:
    :return:
    :rtype:"""
    inp = chw_to_hwc(inp)
    if unnormalise:
        inp *= 255.0
        inp = numpy.clip(inp, 0, 255)
    return inp.astype(numpy.uint8)
