[docs]def frame_generator(
    video_stream: cv2.VideoCapture,
    coder: Optional[Callable] = identity,
) -> Iterable:
    """

    :param video_stream:
    :param coder:"""
    if coder is None:
        coder = identity
    while video_stream.isOpened():
        success, frame = video_stream.read()
        if success:
            yield coder(frame)
        else:
            break
